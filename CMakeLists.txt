cmake_minimum_required(VERSION 3.25)
project(MyTetris C)

set(CMAKE_C_STANDARD 11)

# Create an option to switch between a system sdl library and a vendored sdl library
option(MYGAME_VENDORED "Use vendored libraries" ON)

if (MYGAME_VENDORED)
    add_subdirectory(3rdPart/SDL EXCLUDE_FROM_ALL)
else ()
    # 1. Look for a SDL2 package, 2. look for the SDL2 component and 3. fail if none can be found
    find_package(SDL2 REQUIRED CONFIG REQUIRED COMPONENTS SDL2)

    # 1. Look for a SDL2 package, 2. Look for the SDL2maincomponent and 3. DO NOT fail when SDL2main is not available
    find_package(SDL2 REQUIRED CONFIG COMPONENTS SDL2main)
endif ()

add_executable(MyTetris WIN32 main.c)

# SDL2::SDL2main may or may not be available. It is e.g. required by Windows GUI applications
if (TARGET SDL2::SDL2main)
    # It has an implicit dependency on SDL2 functions, so it MUST be added before SDL2::SDL2 (or SDL2::SDL2-static)
    target_link_libraries(MyTetris PRIVATE SDL2::SDL2main)
endif ()

# Link to the actual SDL2 library. SDL2::SDL2 is the shared SDL library, SDL2::SDL2-static is the static SDL libarary.
target_link_libraries(MyTetris PRIVATE SDL2::SDL2-static)
